def _rebuild_face_overlay():
    global _face_overlay_cache, _face_overlay_dirty
    img = np.zeros((FACE_CANVAS_SIZE, FACE_CANVAS_SIZE, 4), dtype=np.uint8)
    with face_draw_lock:
        strokes = list(face_draw_strokes)
        cur     = face_draw_current
    for stroke in strokes:
        pts = stroke["pts"]
        col = stroke["color"]
        th  = stroke["thickness"]
        for i in range(1, len(pts)):
            cv2.line(img, pts[i-1], pts[i], (*col, 255), th, cv2.LINE_AA)
    if cur and len(cur["pts"]) > 1:
        pts = cur["pts"]
        col = cur["color"]
        th  = cur["thickness"]
        for i in range(1, len(pts)):
            cv2.line(img, pts[i-1], pts[i], (*col, 255), th, cv2.LINE_AA)
    _face_overlay_cache = img
    _face_overlay_dirty = False


def get_face_overlay():
    global _face_overlay_dirty
    if _face_overlay_dirty or _face_overlay_cache is None:
        _rebuild_face_overlay()
    return _face_overlay_cache


def apply_face_overlay(frame, face_landmarks, frame_w, frame_h):
    overlay_bgra = get_face_overlay()
    if overlay_bgra is None:
        return

    lm = face_landmarks.landmark
    S = FACE_CANVAS_SIZE - 1

    forehead  = np.array([lm[10].x * frame_w,  lm[10].y * frame_h],  dtype=np.float32)
    chin      = np.array([lm[152].x * frame_w, lm[152].y * frame_h], dtype=np.float32)
    left_face = np.array([lm[234].x * frame_w, lm[234].y * frame_h], dtype=np.float32)
    right_face= np.array([lm[454].x * frame_w, lm[454].y * frame_h], dtype=np.float32)
    left_eye  = np.array([lm[33].x * frame_w,  lm[33].y * frame_h],  dtype=np.float32)
    right_eye = np.array([lm[263].x * frame_w, lm[263].y * frame_h], dtype=np.float32)
    nose      = np.array([lm[1].x * frame_w,   lm[1].y * frame_h],   dtype=np.float32)

    center     = (forehead + chin) / 2
    face_width = np.linalg.norm(right_face - left_face)
    face_height= np.linalg.norm(chin - forehead)

    x_axis = right_eye - left_eye
    x_axis /= np.linalg.norm(x_axis)
    y_axis = chin - forehead
    y_axis /= np.linalg.norm(y_axis)

    face_mid_x = (left_face[0] + right_face[0]) / 2
    yaw = (nose[0] - face_mid_x) / (face_width / 2)
    yaw = np.clip(yaw, -1.0, 1.0)

    plane_w = face_width * 1.15
    plane_h = face_height * 1.15
    depth   = face_width * 0.18

    tl = center + (-x_axis * plane_w/2) + (-y_axis * plane_h/2)
    tr = center + ( x_axis * plane_w/2) + (-y_axis * plane_h/2)
    br = center + ( x_axis * plane_w/2) + ( y_axis * plane_h/2)
    bl = center + (-x_axis * plane_w/2) + ( y_axis * plane_h/2)

    yaw_push = yaw * depth
    if yaw > 0:
        tr[0] -= yaw_push
        br[0] -= yaw_push
    else:
        tl[0] += yaw_push
        bl[0] += yaw_push

    src_pts = np.float32([[0, 0], [S, 0], [S, S], [0, S]])
    dst_pts = np.float32([tl, tr, br, bl])
    M = cv2.getPerspectiveTransform(src_pts, dst_pts)

    warped = cv2.warpPerspective(
        overlay_bgra, M, (frame_w, frame_h),
        flags=cv2.INTER_LINEAR,
        borderMode=cv2.BORDER_CONSTANT,
        borderValue=(0, 0, 0, 0)
    )

    alpha   = warped[:, :, 3:4].astype(np.float32) / 255.0
    bgr     = warped[:, :, :3].astype(np.float32)
    frame_f = frame.astype(np.float32)
    blended = frame_f * (1 - alpha) + bgr * alpha
    frame[:] = blended.astype(np.uint8)

def _draw_face_preview(canvas_widget, preview_size):
    canvas_widget.delete("face_preview")
    s = preview_size
    fp = _get_canonical_face_points()

    def px(nx, ny):
        return int(nx * s), int(ny * s)

    oval_pts = [px(x, y) for x, y in fp["oval"]]
    flat = [v for pt in oval_pts for v in pt]
    canvas_widget.create_polygon(flat, outline="#aaffaa", fill="", width=1.5,
                                 tags="face_preview")

    for ey in [fp["left_eye"], fp["right_eye"]]:
        cx, cy, rx, ry = ey
        x0, y0 = px(cx - rx, cy - ry)
        x1, y1 = px(cx + rx, cy + ry)
        canvas_widget.create_oval(x0, y0, x1, y1, outline="#aaffaa", fill="",
                                  width=1.5, tags="face_preview")

    for brow in [fp["left_brow"], fp["right_brow"]]:
        bpts = [px(x, y) for x, y in brow]
        for i in range(len(bpts) - 1):
            canvas_widget.create_line(*bpts[i], *bpts[i+1],
                                      fill="#aaffaa", width=1.5,
                                      tags="face_preview")

    npts = [px(x, y) for x, y in fp["nose"]]
    for i in range(len(npts) - 1):
        canvas_widget.create_line(*npts[i], *npts[i+1],
                                  fill="#aaffaa", width=1.5,
                                  tags="face_preview")

    mpts = [px(x, y) for x, y in fp["mouth"]]
    flat_m = [v for pt in mpts for v in pt]
    canvas_widget.create_polygon(flat_m, outline="#aaffaa", fill="",
                                 width=1.5, tags="face_preview")

    corners = [
        (0.10, 0.05, "TL"),
        (0.90, 0.05, "TR"),
        (0.90, 0.97, "BR"),
        (0.10, 0.97, "BL"),
    ]
    for nx, ny, label in corners:
        cx2, cy2 = px(nx, ny)
        canvas_widget.create_oval(cx2 - 4, cy2 - 4, cx2 + 4, cy2 + 4,
                                  outline="#ff9900", fill="#ff9900",
                                  tags="face_preview")
        canvas_widget.create_text(cx2, cy2 - 10, text=label,
                                  fill="#ff9900",
                                  font=("Courier New", 6, "bold"),
                                  tags="face_preview")

def dist(a, b):
    return math.dist(a, b)


def eye_aspect_ratio(lm, idxs, w, h):
    pts = [(int(lm[i].x * w), int(lm[i].y * h)) for i in idxs]
    v1  = dist(pts[1], pts[5])
    v2  = dist(pts[2], pts[4])
    h1  = dist(pts[0], pts[3])
    return (v1 + v2) / (2.0 * h1)


def finger_up(lm, tip, pip):
    return lm[tip].y < lm[pip].y

def detect_gesture(lm):
    index  = finger_up(lm, 8,  6)
    middle = finger_up(lm, 12, 10)
    ring   = finger_up(lm, 16, 14)
    pinky  = finger_up(lm, 20, 18)

    if index and not middle and not ring and not pinky:
        return "paint"
    elif index and middle and not ring and not pinky:
        return "hover"
    elif index and middle and ring and pinky:
        return "erase"
    return "none"


def reset_game():
    global bird_y, velocity, pipes, score, game_over
    bird_y    = 350
    velocity  = 0
    pipes     = []
    score     = 0
    game_over = False

def create_pipe():
    gap_y = random.randint(180, GAME_H - 180)
    pipes.append({"x": GAME_W, "gap_y": gap_y, "passed": False})

def rect_collision(a, b):
    ax, ay, aw, ah = a
    bx, by, bw, bh = b
    return (ax < bx + bw and ax + aw > bx and
            ay < by + bh and ay + ah > by)

def update_game():
    global bird_y, velocity, game_over, score, high_score
    if GAME_PAUSED or game_over:
        return

    velocity += gravity
    bird_y   += velocity

    if bird_y > GAME_H or bird_y < 0:
        game_over = True

    if len(pipes) == 0 or pipes[-1]["x"] < GAME_W - 333:
        create_pipe()

    for pipe in pipes:
        pipe["x"] -= pipe_speed

    pipes[:] = [p for p in pipes if p["x"] > -pipe_width]

    for pipe in pipes:
        top_rect    = (pipe["x"], 0, pipe_width, pipe["gap_y"] - pipe_gap // 2)
        bottom_rect = (pipe["x"], pipe["gap_y"] + pipe_gap // 2, pipe_width, GAME_H)
        bird_rect   = (bird_x - 20, int(bird_y) - 20, 40, 40)

        if rect_collision(bird_rect, top_rect) or rect_collision(bird_rect, bottom_rect):
            game_over = True

        if not pipe["passed"] and pipe["x"] < bird_x:
            pipe["passed"] = True
            score += 1
            if score > high_score:
                high_score = score

    cv2.fillPoly(frame, [pts], (0, 140, 255))


def draw_game():
    game_frame = np.zeros((GAME_H, GAME_W, 3), dtype=np.uint8)
    game_frame[:] = (255, 220, 160)

    pts_mountain = np.array(
        [[0, 700], [0, 480], [120, 390], [250, 490],
         [390, 370], [500, 460], [500, 700]], np.int32)
    cv2.fillPoly(game_frame, [pts_mountain], (165, 140, 130))
    cv2.circle(game_frame, (120, 760), 220, (115, 185, 115), -1)

cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 960)

fps_time = time.time()


FACE_MESH_WIN = "Face Mesh"


def run():
    global blink_count, blink_state, last_point, canvas, velocity, fps_time

    while True:
        success, frame = cap.read()
        if not success:
            break

        frame = cv2.flip(frame, 1)
        h, w, _ = frame.shape
        rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)

        face_results = None
        hand_results = None

        if ENABLE_FACE or ENABLE_EYES or ENABLE_BLINK or ENABLE_FACE_DRAW:
            face_results = face_mesh.process(rgb)

        if ENABLE_HANDS:
            hand_results = hands.process(rgb)

        # ── Face Mesh → webcam frame + dedicated black window ────────────
        if ENABLE_FACE:
            if face_results and face_results.multi_face_landmarks:
                mesh_frame = np.zeros((h, w, 3), dtype=np.uint8)
                for face in face_results.multi_face_landmarks:
                    # Draw on the pure-black separate window
                    mp_draw.draw_landmarks(
                        mesh_frame, face, FACE_TESSELATION,
                        landmark_drawing_spec=None,
                        connection_drawing_spec=mp_draw.DrawingSpec(
                            color=(0, 255, 0), thickness=1)
                    )
                    # Also draw on the live webcam frame
                    mp_draw.draw_landmarks(
                        frame, face, FACE_TESSELATION,
                        landmark_drawing_spec=None,
                        connection_drawing_spec=mp_draw.DrawingSpec(
                            color=(0, 255, 0), thickness=1)
                    )
                # Crop tightly around the face so it fills the small window
                lm0 = face_results.multi_face_landmarks[0].landmark
                xs = [int(p.x * w) for p in lm0]
                ys = [int(p.y * h) for p in lm0]
                pad = 30
                x1 = max(min(xs) - pad, 0)
                y1 = max(min(ys) - pad, 0)
                x2 = min(max(xs) + pad, w)
                y2 = min(max(ys) + pad, h)
                cropped = mesh_frame[y1:y2, x1:x2]
                if cropped.size > 0:
                    ch, cw = cropped.shape[:2]
                    scale = min(400 / cw, 300 / ch)
                    nw, nh = int(cw * scale), int(ch * scale)
                    resized = cv2.resize(cropped, (nw, nh))
                    small = np.zeros((300, 400, 3), dtype=np.uint8)
                    y_off = (300 - nh) // 2
                    x_off = (400 - nw) // 2
                    small[y_off:y_off+nh, x_off:x_off+nw] = resized
                else:
                    small = cv2.resize(mesh_frame, (400, 300))
                cv2.imshow(FACE_MESH_WIN, small)
            else:
                cv2.imshow(FACE_MESH_WIN, np.zeros((300, 400, 3), dtype=np.uint8))
        else:
            try:
                cv2.destroyWindow(FACE_MESH_WIN)
            except cv2.error:
                pass

        # ── Unconnected Eye Mesh (still on webcam frame) ─────────────────
        if ENABLE_EYES and face_results and face_results.multi_face_landmarks:
            lm = face_results.multi_face_landmarks[0].landmark
            for i in EYE_LANDMARKS:
                x_ = int(lm[i].x * w)
                y_ = int(lm[i].y * h)
                cv2.circle(frame, (x_, y_), 2, (255, 255, 0), -1)

        # ── Blink Processing ──────────────────────────────────────────────
        if ENABLE_BLINK and face_results and face_results.multi_face_landmarks:
            lm = face_results.multi_face_landmarks[0].landmark
            left_ear  = eye_aspect_ratio(lm, LEFT_EYE_IDXS, w, h)
            right_ear = eye_aspect_ratio(lm, RIGHT_EYE_IDXS, w, h)
            ear = (left_ear + right_ear) / 2.0

            blinked = False
            if ear < BLINK_THRESHOLD:
                if not blink_state:
                    blink_count += 1
                    blink_state  = True
                    blinked      = True
            else:
                blink_state = False

            cv2.putText(frame, f"Blinks: {blink_count}", (20, 40),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 255), 2)

            if ENABLE_GAME:
                if not game_over and blinked:
                    velocity = jump_strength
                elif game_over and blinked:
                    reset_game()

        # ── Face Drawing Overlay ──────────────────────────────────────────
        if ENABLE_FACE_DRAW and face_results and face_results.multi_face_landmarks:
            apply_face_overlay(frame,
                               face_results.multi_face_landmarks[0],
                               w, h)

        # ── Hands + Paint ─────────────────────────────────────────────────
        if ENABLE_HANDS and hand_results and hand_results.multi_hand_landmarks:
            for hand in hand_results.multi_hand_landmarks:
                lm = hand.landmark
                mp_draw.draw_landmarks(frame, hand, mp_hands.HAND_CONNECTIONS)

                x_ = int(lm[8].x * w)
                y_ = int(lm[8].y * h)
                current_gesture = detect_gesture(lm)

                if ENABLE_PAINT and current_gesture == "paint":
                    if last_point is not None:
                        canvas.append((last_point, (x_, y_)))
                    last_point = (x_, y_)
                elif current_gesture == "hover":
                    last_point = None
                elif current_gesture == "erase":
                    eraser     = (x_, y_)
                    new_canvas = []
                    for p1, p2 in canvas:
                        mx = (p1[0] + p2[0]) // 2
                        my = (p1[1] + p2[1]) // 2
                        if dist((mx, my), eraser) > 35:
                            new_canvas.append((p1, p2))
                    canvas = new_canvas
                    cv2.circle(frame, eraser, 20, (0, 0, 255), 2)
                    last_point = None
                else:
                    last_point = None

        # Draw Paint Canvas
        for p1, p2 in canvas:
            cv2.line(frame, p1, p2, (0, 0, 255), 3)

        # ── FPS ───────────────────────────────────────────────────────────
        current_time = time.time()
        fps      = 1 / (current_time - fps_time)
        fps_time = current_time

        if ENABLE_FPS:
            cv2.putText(frame, f"FPS: {int(fps)}", (20, 80),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)

        # ── Game ──────────────────────────────────────────────────────────
        if ENABLE_GAME:
            update_game()
            draw_game()
        else:
            try:
                cv2.destroyWindow("Flappy Bird AI")
            except cv2.error:
                pass

        cv2.imshow("Webcam Feed", frame)

        key = cv2.waitKey(1)
        if key == 27:   # ESC
            break

    cap.release()
    cv2.destroyAllWindows()

# startup
threading.Thread(target=ui, daemon=True).start()
run()
cap.release()
